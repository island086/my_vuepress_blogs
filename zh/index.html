# * 来自与 ls * (通过通配符匹配 文件名字符(不包括后缀名 .idea 等)为0或多个)

path() {
  # $1 传入的第一个参数
  # $a 取变量a的内容 //或 ${a} "${a}"
  # $() 执行命令 //或 "$()"
  # echo "helloWorld " 输出字符串
  # echo "$a"  输出变量值,(这里的引号，如果没有字符串拼接时，可以不加，加了更好)（双引号以防止通配符和分词）
  # a="helloWorld"  声明变量a,并把helloWorld赋值给a的内容，（推荐->)这里的等号两边不能有空格
  if [[ -d "$1" ]]; then
    echo "目录 $1"

    # ls输出是脆弱的使用 ./* glob 匹配全部文件
    for child in "$1"/*; do
      path "$child"
    done
  else
    echo "文件 $1"
  fi
}
# 获取当前目录
path "$(pwd)"

    restful

    restful定义

    • REST（Representational State Transfer）是一种设计风格，用于构建网络应用程序。它提供了一组规则和指南，帮助开发人员创建可扩展的、结构化的 API（应用程序编程接口）。

    • RESTful API 就是遵循 REST 设计原则构建的 API。这些原则包括：

    • 使用 HTTP 协议的标准方法（例如 GET、PUT、POST 和 DELETE）来操作资源。 使用资源的 URI 来唯一地标识它。 使用网络上的标准协议（例如 HTTP、FTP 等）来传输数据。 使用 XML、JSON 等格式来表示资源。 遵循 REST 设计原则可以使 API 更简单、可扩展、易于理解，也更易于实现。

    sychronized

    1

    • "synchronized" 一词来自希腊语，"syn" 意为 "together"，"chronos" 意为 "time"。因此，"synchronized" 字面意思是 "同时发生" 或 "同步"。在计算机科学领域，"synchronized" 通常指的是多个线程或进程之间的同步，即使它们在不同的时间片内运行，也能够协调它们的行为，从而避免产生冲突。

    2

    • 简而言之，"synchronized" 这个词的翻译为 "同步" 是因为它描述的是多个线程或进程在时间上的协调。
